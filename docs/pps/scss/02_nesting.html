群组 #

scss
div,
p {
  span {
    text-align: center;
  }
}
// 结果如下
div span {
  text-align: center;
}
p span {
  text-align: center;
}
// 另一种情况
div {
  p,
  span {
    text-align: center;
  }
}
// 结果如下
div p {
  text-align: center;
}
div span {
  text-align: center;
}
