属性装饰器 ​

  1. 属性是实例属性时，一参为原型对象
  2. 属性是静态属性时，一参为构造器
  3. 二参为属性名
ts
class Person {
  toProps();
  name = "实例属性";
  static name = "静态属性";
}
function toProps() {
  return (obj: object, name: string) => {};
}
