CSS 预处理器 #

ts
import { defineConfig } from "vite";
import path from "path";
export default defineConfig(({ command, mode }) => {
  return {
    // 路径别名
    resolve: {
      alias: {
        "@": path.resolve(__dirname, "./src"),
      },
    },
    // css预处理器
    css: {
      preprocessorOptions: {
        scss: {
          additionalData: `@use "@/assets/css/index.scss" as *;`,
        },
      },
    },
  };
});

glob 导入 #

ts
const mod = import.meta.glob(
  [
    // 从以下路径导入
    "./mod/**/*.ts",
    // 排除以下
    "!./mod/**/exclude.js",
  ],
  {
    // 是否渴望静态导入
    eager: true,
    // 具名导入
    import: "default",
    // 导入为string
    as: "raw",
  }
);
