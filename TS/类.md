# 类
## 基础
~~~ts
class  Person {
    name:string
    age:number
    constructor(name:string, age:number) {
        this.name = name
        this.age = age
    }
    callName(){
        console.log(this.name)
    }
}
~~~
## 继承
~~~ts
class Animal {
    constructor(age){
        this.age = age
    }
    run(){}
}
class Cat extends Animal {
    constructor(name:string, age:number) {
        super(age)
        this.name = name
    }
    callName(){}
}
~~~
## 属性封装
~~~ts
class Person {
    readonly name = "名字"
    static readonly age:number = 18
    public gender = "公有"
    private privacy = "私有"
    protected protect = "保护"
    get privacy(){
        return this.privacy
    }
    set privacy(value){
        this.privacy = value
    }
}
~~~
