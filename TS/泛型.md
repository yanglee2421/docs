# 泛型
## 在函数中指定
~~~ts
function fun<Typ>(params:Typ) {
    return params
}
~~~
## 指定多个泛型
~~~ts
const fun = <Typ,Syc>()=>{

}
~~~
## 泛型继承类
~~~ts
interface Inter {
    length:number
}
const fun = <typ extends Inter>(a:typ)=>{
    return a.length
}
~~~
## 在类中指定泛型
~~~ts
class Person <Typ> {
    constructor(public parameters:Typ) {
        
    }
}
~~~
